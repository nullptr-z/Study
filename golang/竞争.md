## Channel 通道

> 案例一,10 递减打印到 0

```go
func main() {
	count := 10
	ch := make(chan int, 1)

	go func() {
		<-ch
		for count != 0 {
			count--
			fmt.Println("child:", count)
		}
	}()

	go func() {
		for count > 5 {
			count--
			fmt.Println("main:", count)
		}
		ch <- 1
	}()
}
```

> 案例二，一端写，一端读

```go
func main() {
	ch := make(chan int, 1)
	exitCh := make(chan bool, 1)

	go write_ch(ch)
	go read_ch(ch, exitCh)

	<-exitCh
	fmt.Println("main exit")
}

func write_ch(ch chan int) {
	for i := 0; i < 50; i++ {
		fmt.Println("Write	val:", i)
		ch <- i
	}
	close(ch)
}

func read_ch(ch chan int, exitCh chan bool) {
	for {
		val, ok := <-ch
		if !ok {
			break
		}
		fmt.Println("-Read	val:", val)
	}
	exitCh <- true
}
```

## 互斥锁

虽然简单，但是性能不好，上锁的线程会一直阻塞；这在 go 里面不是一个好的解决方案

```go
func main() {
	var count = 10
	var lock sync.Mutex

	go func() {
		lock.Lock()
		for count != 0 {
			count--
			fmt.Println("child:", count)
		}
		lock.Unlock()
	}()

	lock.Lock()
	for count > 5 {
		time.Sleep(1 * time.Second)
		count--
		fmt.Println("main:", count)
	}
	lock.Unlock()
}
```
